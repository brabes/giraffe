Source code directory layout:

  doc/

  general/
    mlton.mlb
    polyml.sml
    mlton/
    polyml/

  ffi/
    mlton.mlb
    polyml.sml
    mlton/
    polyml/

  auto/                         - generated source code
    mlton.mlb
    polyml.sml

    mlton/                      - C interface files for MLton
      giraffe.h                 - header file generated by MLton from SML
      giraffe-gdk.c
      giraffe-glib.c
      giraffe-gobject.c
      giraffe-gtk.c
      giraffe-vte.c

    polyml/                     - C interface files for Poly/ML
      giraffe-cairo.c
      giraffe-girepository.c
      giraffe-gobject.c
      giraffe-pango.c
      giraffe-gdk.c
      giraffe-glib.c
      giraffe-gtk.c
      giraffe-vte.c

    <namespace>-<version>/      - SML namespace/version-specific directory
      mlton/
      polyml/
      mlton.mlb
      polyml.sml
